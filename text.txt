install nasm with 'brew install nasm'
install cross-binutils with 'brew install i386-elf-binutils i386-elf-gcc'
brew install qemu xorriso
Install grub: brew install i686-elf-grub
Install gdb: brew install gdb

nasm -f elf32 kernel.asm -o build/kasm.o
x86_64-elf-gcc -m32 -ffreestanding -fno-stack-protector -fno-pic -g -c kernel.c -o build/kc.o
x86_64-elf-ld -m elf_i386 -T linker.ld -o build/kernel.bin build/kasm.o build/kc.o
cp build/kernel.bin iso/boot/kernel.bin
i686-elf-grub-mkrescue -o kernel.iso iso

OR WITH GDB DEBUGGING: 
qemu-system-i386 -kernel iso/boot/kernel.bin -S -s
    - F5 on VSCode to start debug session
